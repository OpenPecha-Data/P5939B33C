༄༅། །བདེ་ཆེན་ཞིང་བཀོད་ཀྱི་རྒྱབ་བྱང་དུ།

དུས་གསུམ་རྒྱལ་བ་རྒྱ་མཚོ་ཡི། །ཐུགས་རྗེའི་འོད་ཟེར་འདུས་པའི་སྐུ། །རྣམ་འདྲེན་རྒྱལ་བ་འོད་དཔག་མེད།
།རྣམ་འཕྲུལ་སེམས་དཔའ་སེམས་མ་དང་། །ཉན་ཐོས་དགེ་འདུན་བྱེ་བ་འབུམ། །ཐུགས་རྗེ་མངའ་བ་ཁྱེད་རྣམས་ཀྱི།
།ཡེ་ཤེས་ཆེན་པོས་འདིར་དགོངས་ལ། །ཐུགས་དམ་ཞལ་བཞེས་ཇི་བཞིན་དུ། །བདག་དང་མཐའ་ཡས་སེམས་ཅན་རྣམས།
།འཁོར་བ་ངན་སོང་འཇིགས་ལས་སྐྱོབས། །ཕྱི་ནང་སྡུག་བསྔལ་དབྱིངས་སུ་སོལ། །རང་གཞན་དོན་གཉིས་ལྷུན་གྲུབ་ནས།
།འཆི་མེད་བདེ་བ་ཆེན་པོའི་དབྱིངས། །མགོན་པོ་ཁྱེད་དང་དབྱེར་མེད་མཛོད། །

ཅེས་པའང་མཉན་དཔོན་འཆི་མེད་ངོར། དཔལ་ཀརྨ་པས་བྲིས།།
